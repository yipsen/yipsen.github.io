打开慢 SQL 配置

-- 查看是否开了慢 SQL 配置
show variables like 'show_query%';
show variables like 'long_query_time';

-- 如果没有配置，通过以下语句配置
set global show_query_log='ON'; -- 开启慢 SQL 日志
set global show_query_log_file='/var/lib/mysql/slow-sql.log'; -- 记录慢 SQL 的日志地址
set global long_query_time=1; -- 慢 SQL 阈值，即超过该时间的判定为慢 SQL，单位为秒
